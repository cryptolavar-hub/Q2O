import networkx as nx

def common_neighbors(graph, user):
    return sorted(graph.neighbors(user))

# Example usage
G = nx.Graph()
G.add_edges_from([(1, 2), (1, 3), (2, 3), (2, 4)])
print(common_neighbors(G, 2))  # Output: [1, 3, 4]
```

def jaccard_similarity(graph, user1, user2):
    neighbors1 = set(graph.neighbors(user1))
    neighbors2 = set(graph.neighbors(user2))
    return len(neighbors1.intersection(neighbors2)) / len(neighbors1.union(neighbors2))

# Example usage
print(jaccard_similarity(G, 1, 2))
```
