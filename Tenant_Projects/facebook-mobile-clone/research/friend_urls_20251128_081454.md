```python
def generate_user_url(user_id, user_name):
    base_url = 'https://example.com/users/'
    slug = user_name.replace(' ', '-').lower()
    return f'{base_url}{user_id}-{slug}'

# Example usage
print(generate_user_url(123, 'John Doe'))  # Output: https://example.com/users/123-john-doe
```

```javascript
function generateUserUrl(userId, userName) {
    const baseUrl = 'https://example.com/users/';
    const slug = userName.replace(/\s+/g, '-').toLowerCase();
    return `${baseUrl}${userId}-${slug}`;
}

// Example usage
console.log(generateUserUrl(123, 'Jane Smith')); // Output: https://example.com/users/123-jane-smith
```
