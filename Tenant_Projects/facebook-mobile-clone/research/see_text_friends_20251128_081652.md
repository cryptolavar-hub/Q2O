#### Example 1
**Source**: Example: Friend suggestion using Common Neighbors algorithm
**Language**: python
```python
import networkx as nx

# Create a graph
G = nx.Graph()

# Add edges (friend connections)
G.add_edges_from([(1, 2), (1, 3), (2, 3), (2, 4)])

# Function to suggest friends based on common neighbors

def suggest_friends(graph, user):
    return list(nx.common_neighbors(graph, user, 2))

# Example usage
print(suggest_friends(G, 1))  # Output: [2, 3]
```

from sklearn.metrics import jaccard_score
import numpy as np

# Example user friendships represented as binary vectors
user_a = np.array([1, 1, 0, 1])  # User A's friends
user_b = np.array([1, 0, 1, 1])  # User B's friends

# Calculate Jaccard Index
jaccard_index = jaccard_score(user_a, user_b)
print(f'Jaccard Index: {jaccard_index}')
